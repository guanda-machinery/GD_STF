Examples
此示例顯示瞭如何用 SharedMemory 方法
Copy
class Program
{
   public static Operating Operating = new Operating();
   static void App_Struct()
   {
      //需要使用系統管理員
      GD_STD.Phone.MemoryHelper.OpenSharedMemory(); 
      Operating.OpenApp = true;
      Operating.Satus =PHONE_SATUS.WAIT_MANUA;
      SharedMemory.SetValue<Operating>(Operating);
      Task.Run(() => 
      {
         while (true)
             Operating = SharedMemory.GetValue<Operating>();
      });
      while (true)
      {
         APP_Struct phone = SharedMemory.GetValue<APP_Struct>();//讀取目前記憶體
         phone.OpenOil = true;
         //判斷寫入條件
         if (Operating.Satus == PHONE_SATUS.MANUAL)
         {
             SharedMemory.SetValue<APP_Struct>(phone);
         }
         else
         {
             Console.WriteLine("等待 PC 端允許連線 ........");
         }
   }
}
